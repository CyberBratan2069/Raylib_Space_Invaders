cmake_minimum_required(VERSION 3.16)   # 4.0 doesnâ€™t exist; latest is 3.x
project(SpaceInvaders C)

        src/spaceship.h

            "-framework Cocoa"
            "-framework IOKit"
            "-framework CoreVideo"
    )
endif()

    target_link_libraries(SpaceInvaders m pthread dl GL rt X11)
endif()
